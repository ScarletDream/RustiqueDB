    id INT(32) PRIMARY KEY,
    name VARCHAR(100),
    left_num INT(32),
    discription VARCHAR(150),
    price INT NOT NULL
);

-- 测试5: 多列AND条件
SELECT id, name, price, left_num FROM books_test15
WHERE price > 40 AND price < 70 AND left_num IS NOT NULL;

-- 测试6: 嵌套条件查询
SELECT id, name FROM books_test15
WHERE (id = 5 OR (price > 60 AND price < 80)) AND name = "Data Science";

-- 测试7: 多组括号组合条件
SELECT id, name, price FROM books_test15
WHERE ((id = 1 OR id = 4) AND price > 30) OR (price = 42);

-- 测试8: 多条件组合查询
SELECT id, name, discription, price FROM books_test15
WHERE (price >= 50 OR id <= 2) AND (name = "Rust Programming" OR name = "Web Development");

-- 测试9: 多级嵌套条件
SELECT id, name, price FROM books_test15
WHERE (id = 7 OR (price > 40 AND (name = "Database Design" OR left_num = 10)));

-- 测试10: 复杂组合条件
SELECT id, name, discription, price FROM books_test15
